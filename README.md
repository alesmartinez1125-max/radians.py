# radians.py this will focus on radians and radians connected math functions 
# where as a pie chart focuses on the whole, measured by individual pieces, radians measure the outer border 

Import math as m 

# before I give an example of the radians, I will demonstrate the cos function. cosine measures horizontal distance from the center

print(m.cos(2))
  output -0.4161468365471424

print(m.cos(m.radians(2)))
  output 09993908270190958

# again I will given an example of the connecting math function first. Sine measures verticval height from the center 

print(m.sin(7))
  output 0.6569865987187891

print(m.sin(m.radians(7)))
  output 0.12186934340514748

# the final section will follow the same example pattern. Tangent measures the ratio height to width(Slope)

print(m.tan(3))
  output 0.1425465430742778

print(m.tan(m.radians(3)))
  output 0.05240777928304121
